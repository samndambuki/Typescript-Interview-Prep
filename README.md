# Create Project from start

# Create new directory e.g

my-typescript-project

# Generate package.json

npm init -y

# Install typescript

npm install typescript --save-dev

# Check whether typescript installed

npx tsc --version

# Create ts config file

npx tsc --init

# Create a source folder

src/index.ts

# Compile project project

npx tsc

# Run compiled prject

node dist/index.js
